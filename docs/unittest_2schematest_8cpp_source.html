113#define VALIDATE(schema, json, expected) \
114{\
115 VALIDATE_(schema, json, expected, true) \
116}
117
118#define VALIDATE_(schema, json, expected, expected2) \
119{\
120 EXPECT_TRUE(expected2 == schema.GetError().ObjectEmpty());\
121 EXPECT_TRUE(schema.IsSupportedSpecification());\
122 SchemaValidator validator(schema);\
123 Document d;\
124 /*printf("\n%s\n", json);*/\
125 d.Parse(json);\
126 EXPECT_FALSE(d.HasParseError());\
127 EXPECT_TRUE(expected == d.Accept(validator));\
128 EXPECT_TRUE(expected == validator.IsValid());\
129 ValidateErrorCode code = validator.GetInvalidSchemaCode();\
130 if (expected) {\
131 EXPECT_TRUE(code == kValidateErrorNone);\
132 EXPECT_TRUE(validator.GetInvalidSchemaKeyword() == 0);\
133 }\
134 if ((expected) && !validator.IsValid()) {\
135 StringBuffer sb;\
136 validator.GetInvalidSchemaPointer().StringifyUriFragment(sb);\
137 printf("Invalid schema: %s\n", sb.GetString());\
138 printf("Invalid keyword: %s\n", validator.GetInvalidSchemaKeyword());\
139 printf("Invalid code: %d\n", code);\
140 printf("Invalid message: %s\n", GetValidateError_En(code));\
141 sb.Clear();\
142 validator.GetInvalidDocumentPointer().StringifyUriFragment(sb);\
143 printf("Invalid document: %s\n", sb.GetString());\
144 sb.Clear();\
145 Writer<StringBuffer> w(sb);\
146 validator.GetError().Accept(w);\
147 printf("Validation error: %s\n", sb.GetString());\
148 }\
149}

151#define INVALIDATE(schema, json, invalidSchemaPointer, invalidSchemaKeyword, invalidDocumentPointer, error) \
152{\
153 INVALIDATE_(schema, json, invalidSchemaPointer, invalidSchemaKeyword, invalidDocumentPointer, error, kValidateDefaultFlags, SchemaValidator, Pointer) \
154}
155
156#define INVALIDATE_(schema, json, invalidSchemaPointer, invalidSchemaKeyword, invalidDocumentPointer, error, \
157 flags, SchemaValidatorType, PointerType) \
158{\
159 EXPECT_TRUE(schema.GetError().ObjectEmpty());\
160 EXPECT_TRUE(schema.IsSupportedSpecification());\
161 SchemaValidatorType validator(schema);\
162 validator.SetValidateFlags(flags);\
163 Document d;\
164 /*printf("\n%s\n", json);*/\
165 d.Parse(json);\
166 EXPECT_FALSE(d.HasParseError());\
167 d.Accept(validator);\
168 EXPECT_FALSE(validator.IsValid());\
169 ValidateErrorCode code = validator.GetInvalidSchemaCode();\
170 ASSERT_TRUE(code != kValidateErrorNone);\
171 ASSERT_TRUE(strcmp(GetValidateError_En(code), "Unknown error.") != 0);\
172 if (validator.GetInvalidSchemaPointer() != PointerType(invalidSchemaPointer)) {\
173 StringBuffer sb;\
174 validator.GetInvalidSchemaPointer().Stringify(sb);\
175 printf("GetInvalidSchemaPointer() Expected: %s Actual: %s\n", invalidSchemaPointer, sb.GetString());\
176 ADD_FAILURE();\
177 }\
178 ASSERT_TRUE(validator.GetInvalidSchemaKeyword() != 0);\
179 if (strcmp(validator.GetInvalidSchemaKeyword(), invalidSchemaKeyword) != 0) {\
180 printf("GetInvalidSchemaKeyword() Expected: %s Actual %s\n", invalidSchemaKeyword, validator.GetInvalidSchemaKeyword());\
181 ADD_FAILURE();\
182 }\
183 if (validator.GetInvalidDocumentPointer() != PointerType(invalidDocumentPointer)) {\
184 StringBuffer sb;\
185 validator.GetInvalidDocumentPointer().Stringify(sb);\
186 printf("GetInvalidDocumentPointer() Expected: %s Actual: %s\n", invalidDocumentPointer, sb.GetString());\
187 ADD_FAILURE();\
188 }\
189 Document e;\
190 e.Parse(error);\
191 if (validator.GetError() != e) {\
192 StringBuffer sb;\
193 Writer<StringBuffer> w(sb);\
194 validator.GetError().Accept(w);\
195 printf("GetError() Expected: %s Actual: %s\n", error, sb.GetString());\
196 ADD_FAILURE();\
197 }\
198}

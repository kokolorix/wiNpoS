275

277 }
278 }

281 {

292 }
293 break;
294 case WM_PAINT:
295 {
296 PAINTSTRUCT ps;
297 HDC hdc = BeginPaint(hWnd, &ps);

367void AttachToProcess(ULONG targetPid)
368{
369 // Open the target process with read/write access
370 HANDLE hProcess = OpenProcess(PROCESS_ALL_ACCESS, FALSE, targetPid);
371 if (hProcess == NULL)
372 return Utils::ShowLastError("Failed to open process, error {}");
373
374 // Get the address of the LoadLibrary function in the kernel32.dll module
375 HMODULE hKernel32 = GetModuleHandle(_T("kernel32.dll"));
376 if (hKernel32 == NULL)
377 return Utils::ShowLastError("Failed to obtain kernel32.dll handle, error {}");
378
379 FARPROC pLoadLibrary = GetProcAddress(hKernel32, "LoadLibraryW");
380 if (pLoadLibrary == nullptr)
381 return Utils::ShowLastError("Failed to obtain LoadLibraryW adress, error {}");
382
383 WCHAR path[MAX_PATH];
384 DWORD pathLength = GetModuleFileName(NULL, path, MAX_PATH);
385 PathRemoveFileSpec(path);
386#ifdef _WIN64
387 PathAppend(path, L"wiNpoS-Hook64.dll");
388#else
389 PathAppend(path, L"wiNpoS-Hook32.dll");
390#endif // _WIN64
391
392 // Allocate memory in the target process to hold the path to the DLL
393 SIZE_T pathSize = (_tcslen(path) * sizeof(_TCHAR));
394 LPVOID remotePath = VirtualAllocEx(hProcess, NULL, pathSize, MEM_COMMIT, PAGE_READWRITE);
395 if (remotePath == NULL)
396 {
397 Utils::ShowLastError("Failed to allocate memory in process, error {}");
398 CloseHandle(hProcess);
399 return;
400 }
401
402 // Write the path to the DLL to the allocated memory in the target process
403 if (!WriteProcessMemory(hProcess, remotePath, path, pathSize, NULL))
404 {
405 Utils::ShowLastError("Failed to write to process memory, error {}");
406 VirtualFreeEx(hProcess, remotePath, 0, MEM_RELEASE);
407 CloseHandle(hProcess);
408 return;
409 }
410
411 // Create a remote thread in the target process to call LoadLibrary with the path to the DLL
412 HANDLE hThread = CreateRemoteThread(hProcess, NULL, 0, (LPTHREAD_START_ROUTINE)pLoadLibrary, remotePath, 0, NULL);
413 if (hThread == NULL)
414 {
415 Utils::ShowLastError("Failed to create remote thread, error {}");
416 VirtualFreeEx(hProcess, remotePath, 0, MEM_RELEASE);
417 CloseHandle(hProcess);
418 return;
419 }
420
421 // Wait for the remote thread to finish executing
422 WaitForSingleObject(hThread, INFINITE);
423
424 // Free the memory and close the handles
425 VirtualFreeEx(hProcess, remotePath, 0, MEM_RELEASE);
426 CloseHandle(hThread);
427 CloseHandle(hProcess);
428}

430void DrawWindowBorderForTargeting(_In_ HWND hWnd)
431{
432 RECT rect;
433 HDC hdc;
434
435 GetWindowRect(hWnd, &rect);
436 hdc = GetWindowDC(hWnd);
437
438 if (hdc)
439 {
440 INT penWidth;
441 INT oldDc;
442 HPEN pen;
443 HBRUSH brush;
444
445 penWidth = GetSystemMetrics(SM_CXBORDER) * 3;
446
447 oldDc = SaveDC(hdc);
448
449 // Get an inversion effect.
450 SetROP2(hdc, R2_NOT);
451
452 pen = CreatePen(PS_INSIDEFRAME, penWidth, RGB(0x00, 0x00, 0x00));
453 SelectPen(hdc, pen);
454
455 brush = GetStockBrush(NULL_BRUSH);
456 SelectBrush(hdc, brush);
457
458 // Draw the rectangle.
459 Rectangle(hdc, 0, 0, rect.right - rect.left, rect.bottom - rect.top);
460
461 // Cleanup.
462 DeletePen(pen);
463
464 RestoreDC(hdc, oldDc);
465 ReleaseDC(hWnd, hdc);
466 }

114LRESULT CALLBACK GetMessageHookProc(int nCode, WPARAM wParam, LPARAM lParam)
115{
116 switch (nCode)
117 {
118 case HC_ACTION:
119 {
120 if (wParam == PM_REMOVE)
121 {
122 MSG* pMsg = (MSG*)lParam;
123
124 //WRITE_DEBUG_LOG(format("Msg: {}", pMsg->message));
125
126 if (pMsg->message == MT_HOOK_MSG_REGISTER_THREAD_HOOK)
127 {
128 hooks.setHooks(hInstance);
129 }
130
131 else if (pMsg->message == MT_HOOK_MSG_CREATE_TASK_TOOLBAR)
132 {
133 HRESULT hr = thumbnailToolbar.initialize(hInstance, pMsg->hwnd);
134 if (hr == 0x800401f0)//0x800401f0 : CoInitialize has not been called.
135 {
136 coInit = CoInitialize(NULL);
137 hr = thumbnailToolbar.initialize(hInstance, pMsg->hwnd);
138 }
139 if (FAILED(hr))
140 WRITE_DEBUG_LOG(format("Initialize of task toolbar failed with {:#010x} for {}", hr, Utils::ExeName));
141 else
142 WRITE_DEBUG_LOG(format("Initialize of task toolbar succede with {:#010x} for {}", hr, Utils::ExeName));
143
144 uint32_t sourceThreadId = (uint32_t)pMsg->wParam;
145 if(sourceThreadId != GetCurrentThreadId())
146 PostThreadMessage(sourceThreadId, WM_QUIT, 0, 0);
147 }
148
149 else if (pMsg->message == MT_HOOK_MSG_UNLOAD)
150 {
151 WRITE_DEBUG_LOG(format("Msg: {}(MT_HOOK_MSG_UNLOAD) in {}", pMsg->message, Utils::ExeName));
152
153 if(pMsg->wParam != GetCurrentProcessId())
154 {
155 std::thread t([]() {
156 WRITE_DEBUG_LOG(format("Unload {} from {}", Utils::DllName, Utils::ExeName));
157 FreeLibrary(hInstance);
158 });
159
160 t.detach(); // because the end of control is reached here
161 }
162 }
163 }
164 }
165 }
166
167 return CallNextHookEx(NULL, nCode, wParam, lParam);
